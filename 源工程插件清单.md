#### 源工程已有插件，如需依赖插件开发调试可用cordova命令添加
```
 <!-- 访问外部链接 -->
 "cordova-plugin-whitelist": "1.3.3",
 命令：cordova plugin add cordova-plugin-whitelist

 <!-- 调用设备上的摄像头 -->
 "cordova-plugin-camera": "4.0.2",
 命令：cordova plugin add cordova-plugin-camera

 <!-- 获取设备上联系人的信息 -->
 "cordova-plugin-contacts": "3.0.1",
 命令：cordova plugin add cordova-plugin-contacts

 <!-- 获取设备信息 -->
 "cordova-plugin-device": "2.0.1",
 命令：cordova plugin add cordova-plugin-device

 <!-- 在设备上读/写 -->
 "cordova-plugin-file": "6.0.1",
 命令：cordova plugin add cordova-plugin-file

 <!-- 文件上传或者下载 -->
 "cordova-plugin-file-transfer": "1.7.1",
 命令：cordova plugin add cordova-plugin-file-transfer

 <!-- 打开新的浏览器窗口（设备默认浏览器） -->
 "cordova-plugin-inappbrowser": "2.0.2",
 命令：cordova plugin add cordova-plugin-inappbrowser

 <!-- 获取网络连接信息 -->
 "cordova-plugin-network-information": "2.0.1",
 命令：cordova plugin add cordova-plugin-network-information

 <!-- 解压缩文件 -->
 "cordova-plugin-zip": "3.1.0",
 命令：cordova plugin add https://github.com/MobileChromeApps/zip.git

 <!-- 获取当前应用的版本号 -->
 "cordova-plugin-app-version": "0.1.9",
 命令：cordova plugin add cordova-plugin-app-version

 <!-- 二维码扫描和创建 -->
 "cordova-plugin-barcodescanner": "0.7.9",
 命令：cordova plugin add phonegap-plugin-barcodescanner

 <!-- 图片选择 -->
 "cordova-plugin-image-picker": "1.1.1",
 命令：cordova plugin add cordova-plugin-image-picker

 <!-- 提示弹窗 -->
 "cordova-plugin-x-toast": "2.6.0"
 命令：cordova plugin add cordova-plugin-x-toast

```
